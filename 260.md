# leetcode 206 题

## 方法一 使用 map 额外空间

```js
var singleNumber = function (nums) {
  // 申请一个额外空间
  let obj = {}
  //遍历数组
  nums.forEach((item) => {
    //如果当前值在空间中找不到
    if (obj[item] == undefined) {
      //记录当前值出现的次数
      obj[item] = 1
    } else {
      // 再次出现的删除
      delete obj[item]
    }
  })

  //将数据从额外空间提取出来
  let result = Object.keys(obj).map((v) => Number(v))
  return result
  console.log(obj)
}
```
